// Create a tooltip element for displaying summaries
function initTooltip() {
  const tooltip = document.createElement('div');
  tooltip.id = 'wiki-summary-tooltip';
  tooltip.style.cssText = `
    position: fixed;
    background: linear-gradient(135deg, #ffffff 0%, #f8f9fa 100%);
    border: 1px solid #e0e0e0;
    border-radius: 12px;
    padding: 0;
    max-width: 380px;
    box-shadow: 0 10px 30px rgba(0, 0, 0, 0.15);
    z-index: 10000;
    font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
    display: none;
    cursor: default;
    overflow: hidden;
  `;
  document.body.appendChild(tooltip);
  return tooltip;
}

let tooltip = null;

// Wait for DOM to be ready
if (document.body) {
  tooltip = initTooltip();
} else {
  document.addEventListener('DOMContentLoaded', () => {
    tooltip = initTooltip();
  });
}

// Listen for text selection
document.addEventListener('mouseup', () => {
  if (!tooltip) return;
  
  const selectedText = window.getSelection().toString().trim();
  
  if (selectedText.length > 2) {
    const range = window.getSelection().getRangeAt(0);
    const rect = range.getBoundingClientRect();
    
    // Position tooltip near the selected text with screen boundary check
    let top = rect.bottom + window.scrollY + 10;
    let left = rect.left + window.scrollX;
    
    // Adjust if tooltip would go off-screen
    if (left + 380 > window.innerWidth) {
      left = window.innerWidth - 390;
    }
    if (left < 10) {
      left = 10;
    }
    
    tooltip.style.top = top + 'px';
    tooltip.style.left = left + 'px';
    
    // Show loading state
    tooltip.style.display = 'block';
    tooltip.innerHTML = `
      <div style="padding: 20px; text-align: center;">
        <div style="display: inline-block; width: 30px; height: 30px; border: 3px solid #e0e0e0; border-top: 3px solid #3366cc; border-radius: 50%; animation: spin 1s linear infinite;"></div>
        <p style="margin: 10px 0 0 0; color: #666; font-size: 13px;">Fetching summary...</p>
      </div>
      <style>
        @keyframes spin {
          0% { transform: rotate(0deg); }
          100% { transform: rotate(360deg); }
        }
      </style>
    `;
    
    // Request summary from background script
    chrome.runtime.sendMessage(
      { action: 'getWikiSummary', query: selectedText },
      (response) => {
        if (response && response.success) {
          displaySummary(response.data);
        } else {
          tooltip.innerHTML = '<div style="padding: 15px; text-align: center; color: #d9534f; font-size: 13px;">Could not find summary</div>';
        }
      }
    );
  } else {
    tooltip.style.display = 'none';
  }
});

// Hide tooltip when clicking elsewhere
document.addEventListener('mousedown', (e) => {
  if (tooltip && !tooltip.contains(e.target)) {
    tooltip.style.display = 'none';
  }
});

function displaySummary(data) {
  const { title, extract, url, thumbnail } = data;
  
  const thumbnailHtml = thumbnail ? `
    <img src="${thumbnail}" alt="${title}" style="width: 100%; height: 150px; object-fit: cover; display: block;">
  ` : '';
  
  tooltip.innerHTML = `
    ${thumbnailHtml}
    <div style="padding: 16px;">
      <div style="display: flex; justify-content: space-between; align-items: start; gap: 10px;">
        <h3 style="margin: 0 0 10px 0; color: #1a1a1a; font-size: 15px; font-weight: 600; flex: 1;">${title}</h3>
        <button id="wiki-close-btn" style="
          background: none;
          border: none;
          color: #999;
          font-size: 18px;
          cursor: pointer;
          padding: 0;
          width: 24px;
          height: 24px;
          display: flex;
          align-items: center;
          justify-content: center;
          flex-shrink: 0;
        ">✕</button>
      </div>
      <p style="margin: 0 0 12px 0; color: #555; font-size: 13px; line-height: 1.5;">${extract}</p>
      <a id="wiki-read-btn" href="${url}" target="_blank" style="
        display: inline-block;
        color: #fff;
        background: #3366cc;
        text-decoration: none;
        font-size: 12px;
        font-weight: 500;
        padding: 8px 14px;
        border-radius: 6px;
        transition: background 0.2s;
        cursor: pointer;
      ">
        Read on Wikipedia →
      </a>
    </div>
  `;
  
  // Add close button listener
  const closeBtn = document.getElementById('wiki-close-btn');
  if (closeBtn) {
    closeBtn.addEventListener('click', (e) => {
      e.preventDefault();
      e.stopPropagation();
      tooltip.style.display = 'none';
    });
  }
  
  // Add hover effects to read button
  const readBtn = document.getElementById('wiki-read-btn');
  if (readBtn) {
    readBtn.addEventListener('mouseover', () => {
      readBtn.style.background = '#254fa0';
    });
    readBtn.addEventListener('mouseout', () => {
      readBtn.style.background = '#3366cc';
    });
  }
}
